## 一、核心逻辑验证

### 1. ID 归一化测试

**方法 A：浏览器控制台测试**

```javascript
// 打开浏览器控制台 (F12)，执行以下代码
import { normalizeId } from './src/utils/problemLogic.js'

// 测试用例
console.log(normalizeId('M01_001'))  // 期望: 'm01-001'
console.log(normalizeId('m01-001'))  // 期望: 'm01-001'
console.log(normalizeId('M01-001'))  // 期望: 'm01-001'
console.log(normalizeId('M01-001  '))  // 期望: 'm01-001' (trim)
console.log(normalizeId(null))  // 期望: null
console.log(normalizeId(''))  // 期望: null
```

**方法 B：创建测试文件**

```javascript
// src/utils/__tests__/problemLogic.test.js
import { describe, it, expect } from 'vitest'
import { normalizeId } from '../problemLogic'

describe('normalizeId', () => {
  it('should normalize underscore to dash', () => {
    expect(normalizeId('M01_001')).toBe('m01-001')
  })

  it('should convert to lowercase', () => {
    expect(normalizeId('M01-001')).toBe('m01-001')
  })

  it('should trim whitespace', () => {
    expect(normalizeId('  m01-001  ')).toBe('m01-001')
  })

  it('should return null for invalid input', () => {
    expect(normalizeId(null)).toBe(null)
    expect(normalizeId('')).toBe(null)
    expect(normalizeId(123)).toBe(null)
  })
})
```

### 2. 难度边界测试

**方法 A：浏览器控制台测试**

```javascript
import { getDifficultyByElo } from './src/utils/problemLogic.js'

// 边界测试
```

**方法 B：测试文件**

```javascript
// src/utils/__tests__/problemLogic.test.js
import { getDifficultyByElo } from '../problemLogic'

describe('getDifficultyByElo', () => {
  it('should return L2 for elo <= 1800', () => {
    expect(getDifficultyByElo(0).level).toBe('L2')
    expect(getDifficultyByElo(1800).level).toBe('L2')
  })

  it('should return L3 for 1800 < elo <= 2500', () => {
    expect(getDifficultyByElo(1801).level).toBe('L3')
    expect(getDifficultyByElo(2500).level).toBe('L3')
  })

  it('should return L4 for elo > 2500', () => {
    expect(getDifficultyByElo(2501).level).toBe('L4')
    expect(getDifficultyByElo(3000).level).toBe('L4')
  })
})
```

### 4. 加权随机测试

**方法 A：浏览器控制台测试**

```javascript
import { selectVariableKnobs } from './src/utils/problemLogic.js'

// 运行 1000 次统计分布
const pool = [
  { name: "基础整数参数", weight: 3 },
  { name: "含根式参数", weight: 2 },
  { name: "分数参数", weight: 1 }
]

const counts = {}
for (let i = 0; i < 1000; i++) {
  const selected = selectVariableKnobs(pool)
  counts[selected.name] = (counts[selected.name] || 0) + 1
}

console.log('分布统计:', counts)
// 期望: 基础整数参数 ~500, 含根式参数 ~333, 分数参数 ~167
// 比例约为 3:2:1
```

### 3. 单题失败隔离测试

**方法：代码模拟**

在 `handleGenerateBundle` 中模拟失败：

```javascript
const promises = selectedTasks.map(async (task) => {
  try {
    // 模拟某道题失败
    if (task.motifId === 'M01-001') {
      throw new Error('模拟失败')
    }
    // 正常生成
    const variant = await generateSingleProblem(...)
    return { ...task, variant }
  } catch (err) {
    console.error(`生成任务 ${task.motifId} 失败:`, err)
    return null  // 返回 null，不影响其他题目
  }
})
```

**期望结果**：
- 其他题目正常显示
- 控制台显示失败日志
